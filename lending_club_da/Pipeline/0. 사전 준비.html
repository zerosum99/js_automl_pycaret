from pyspark.sql import SparkSession
from pyspark.conf import SparkConf
import os 


def set_spark():
    os.environ['HADOOP_USER_NAME']='dpcore'

    settings = [
        #jdbc 접속정보
        ('spark.jars.packages', 'org.xerial:sqlite-jdbc:3.34.0'),
        #S3 접속
        ("spark.hadoop.fs.s3a.impl", "org.apache.hadoop.fs.s3a.S3AFileSystem"),
        ("spark.driver.extraClassPath",f"{AWS_JAVA_SDK}:{AWS_JAVA_SDK_S3}:{AWS_JAVA_SDK_CORE}:{HADOOP_AWS_AMZN}")
    ]
    conf = SparkConf().setAppName('data ingestion').setAll(settings)
    spark= SparkSession.builder\
           .config(conf=conf)\
           .getOrCreate()
    
    return spark

def get_df_from_table(table):
    df= spark.read.format('jdbc') \
        .options(driver='org.sqlite.JDBC', dbtable=table,
                 url=URL)\
        .load()
    return df

def sava_parquet_from_table_to_S3(table):
    df = get_df_from_table(table)
    table_dir = f'{S3_BUCKET_DIR}{S3_DATA_DIR}{table}'
    df.write.option("header", "true").mode("overwrite").parquet(table_dir)
    print(f"TABLE NAME : {table}")
    print(table_dir)

import boto3

def upload_model(model):
    s3_client = boto3.client('s3',
                aws_access_key_id=S3_ACCESS_KEY,
                aws_secret_access_key=S3_SECRET_KEY)
    response = s3_client.upload_file(LOCAL_MODEL_DIR+model, BUCKET_NAME, S3_MODEL_DIR+model)
    print(f"MODEL NAME : {model}")
    print(S3_MODEL_DIR+model)
